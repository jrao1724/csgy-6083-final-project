-- Create the EnergyPrices table
CREATE TABLE EnergyPrices (
  PRIMARY KEY (priceTimeID, zipCode),
  priceTimeID INT,
  zipCode VARCHAR(10),
  timePeriodStart VARCHAR(50),
  timePeriodEnd VARCHAR(50),
  price DECIMAL(10, 2)
);

-- Create the EnergyConsumption table
CREATE TABLE EnergyConsumption (
  PRIMARY KEY (eventTimestamp, deviceID),
  FOREIGN KEY (deviceID) REFERENCES Devices(deviceID),
  FOREIGN KEY (priceTimeID) REFERENCES EnergyPrices(priceTimeID),
  eventTimestamp TIMESTAMP,
  eventLabel VARCHAR(255),
  numValue INT,
  totalEnergyConsumed DECIMAL(10, 2),
  deviceID INT,
  priceTimeID INT
);
